vn100_windows_dynamic_dll.c

This example is a conversion of the vn100_windows_basic.c so that it uses a dynamically loaded DLL provided by the library. The Windows API is used via the windows.h header file to load the DLL into memory using the function LoadLibrary and then pointers to the functions provided by the DLL are retrieved individually using the function GetProcAddress.

/* NOTE: Make sure you define the preprocessor directive EXPORT_TO_DLL for your
* project so that when you include the file vectornav.h, you will use the C-style
* function calling convention. */
#include <tchar.h>
#include <stdio.h>
#include "vectornav.h"
/* Include the Window API to get access to the DLL loading functions. */
#include <windows.h>
/* Change the connection settings to your configuration. */
const char* const COM_PORT = "COM1";
const int BAUD_RATE = 115200;
/* We typedef the signatures of the fuctions we want to use from the DLL. These
signatures can be copied from either the vn100.h or vn100.h files. */
typedef VN_ERROR_CODE (*VN100CONNECTFUNC)(Vn100*, const char*, int);
typedef VN_ERROR_CODE (*VN100GETYAWPITCHROLLFUNC)(Vn100*, VnYpr*);
typedef VN_ERROR_CODE (*VN100DISCONNECTFUNC)(Vn100*);
int _tmain(int argc, _TCHAR* argv[])
{
VnYpr ypr;
Vn100 vn100;
VN_ERROR_CODE errorCode;
int i;
/* Variables required for handling the DLL functions. */
HINSTANCE hDLL;
VN100CONNECTFUNC vn100_connect_ptr;
VN100DISCONNECTFUNC vn100_disconnect_ptr;
VN100GETYAWPITCHROLLFUNC vn100_getYawPitchRoll_ptr;
/* First load the DLL into memory and get a handle to it. The file
* VectorNav.CCpp.dll must be located in the same location where the
* executable is running. */
hDLL = LoadLibrary("VectorNav.CCpp.dll");
/* NOTE: The code snippet below is useful for determining what system
* errors are encountered when trying to load the DLL libraries.
DWORD lastError;
lastError = GetLastError(); */
/* Now let's get a pointer to each function in the DLL we wish to call. */
vn100_connect_ptr = (VN100CONNECTFUNC) GetProcAddress(hDLL, "vn100_connect");
vn100_disconnect_ptr = (VN100DISCONNECTFUNC) GetProcAddress(hDLL, "vn100_disconnect");
vn100_getYawPitchRoll_ptr = (VN100GETYAWPITCHROLLFUNC) GetProcAddress(hDLL, "vn100_getYawPitchRoll");
/* Notice we are using our function pointer here. */
errorCode = vn100_connect_ptr(&vn100, COM_PORT, BAUD_RATE);
for (i = 0; i < 10; i++) {
/* Function pointer used here too. */
errorCode = vn100_getYawPitchRoll_ptr(&vn100, &ypr);
printf("YPR: %+#7.2f %+#7.2f %+#7.2f\n", ypr.yaw, ypr.pitch, ypr.roll);
Sleep(1000);
}
/* DLL function pointer used here also. */
vn100_disconnect_ptr(&vn100);
/* Now release the DLL library since we are done using it. */
FreeLibrary(hDLL);
return 0;
}
