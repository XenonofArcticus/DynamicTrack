vn200_windows_dynamic_dll.c

This example is a conversion of the vn200_windows_basic.c so that it uses a dynamically loaded DLL provided by the library. The Windows API is used via the windows.h header file to load the DLL into memory using the function LoadLibrary and then pointers to the functions provided by the DLL are retrieved individually using the function GetProcAddress.

/* NOTE: Make sure you define the preprocessor directive EXPORT_TO_DLL for your
* project so that when you include the file vectornav.h, you will use the C-style
* function calling convention. */
#include <tchar.h>
#include <stdio.h>
#include "vectornav.h"
/* Include the Window API to get access to the DLL loading functions. */
#include <windows.h>
/* Change the connection settings to your configuration. */
const char* const COM_PORT = "COM1";
const int BAUD_RATE = 115200;
/* We typedef the signatures of the fuctions we want to use from the DLL. These
signatures can be copied from either the vn100.h or vn200.h files. */
typedef VN_ERROR_CODE (*VN200CONNECTFUNC)(Vn200*, const char*, int);
typedef VN_ERROR_CODE (*VN200GETINSSOLUTIONFUNC)(Vn200*, double*, unsigned short*, unsigned short*, VnVector3*, VnVector3*, VnVector3*, float*, float*, float*);
typedef VN_ERROR_CODE (*VN200DISCONNECTFUNC)(Vn200*);
int _tmain(int argc, _TCHAR* argv[])
{
double gpsTime;
unsigned short gpsWeek, status;
VnVector3 ypr, latitudeLognitudeAltitude, nedVelocity;
float attitudeUncertainty, positionUncertainty, velocityUncertainty;
VN_ERROR_CODE errorCode;
Vn200 vn200;
int i;
/* Variables required for handling the DLL functions. */
HINSTANCE hDLL;
VN200CONNECTFUNC vn200_connect_ptr;
VN200DISCONNECTFUNC vn200_disconnect_ptr;
VN200GETINSSOLUTIONFUNC vn200_getInsSolution_ptr;
/* First load the DLL into memory and get a handle to it. The file
* VectorNav.CCpp.dll must be located in the same location where the
* executable is running. */
hDLL = LoadLibrary("VectorNav.CCpp.dll");
/* NOTE: The code snippet below is useful for determining what system
* errors are encountered when trying to load the DLL libraries.
DWORD lastError;
lastError = GetLastError(); */
/* Now let's get a pointer to each function in the DLL we wish to call. */
vn200_connect_ptr = (VN200CONNECTFUNC) GetProcAddress(hDLL, "vn200_connect");
vn200_disconnect_ptr = (VN200DISCONNECTFUNC) GetProcAddress(hDLL, "vn200_disconnect");
vn200_getInsSolution_ptr = (VN200GETINSSOLUTIONFUNC) GetProcAddress(hDLL, "vn200_getInsSolution");
/* Notice we are using our function pointer here. */
errorCode = vn200_connect_ptr(&vn200, COM_PORT, BAUD_RATE);
for (i = 0; i < 10; i++) {
/* Function pointer used here too. */
errorCode = vn200_getInsSolution_ptr(
&vn200,
&gpsTime,
&gpsWeek,
&status,
&ypr,
&latitudeLognitudeAltitude,
&nedVelocity,
&attitudeUncertainty,
&positionUncertainty,
&velocityUncertainty);
printf("INS Solution:\n"
" GPS Time: %f\n"
" GPS Week: %u\n"
" INS Status: %.4X\n"
" YPR.Yaw: %+#7.2f\n"
" YPR.Pitch: %+#7.2f\n"
" YPR.Roll: %+#7.2f\n"
" LLA.Lattitude: %+#7.2f\n"
" LLA.Longitude: %+#7.2f\n"
" LLA.Altitude: %+#7.2f\n"
" Velocity.North: %+#7.2f\n"
" Velocity.East: %+#7.2f\n"
" Velocity.Down: %+#7.2f\n"
" Attitude Uncertainty: %+#7.2f\n"
" Position Uncertainty: %+#7.2f\n"
" Velocity Uncertainty: %+#7.2f\n",
gpsTime,
gpsWeek,
status,
ypr.c0,
ypr.c1,
ypr.c2,
latitudeLognitudeAltitude.c0,
latitudeLognitudeAltitude.c1,
latitudeLognitudeAltitude.c2,
nedVelocity.c0,
nedVelocity.c1,
nedVelocity.c2,
attitudeUncertainty,
positionUncertainty,
velocityUncertainty);
printf("\n\n");
Sleep(1000);
}
/* DLL function pointer used here also. */
vn200_disconnect_ptr(&vn200);
/* Now release the DLL library since we are done using it. */
FreeLibrary(hDLL);
return 0;
}
